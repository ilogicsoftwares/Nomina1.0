SELECT PRENOMINAGEN.*, 
NOMINAUNI.DESDE,
NOMINAUNI.HASTA,
TRABAJADOR.*,
CARGO.NOMBRE AS CARGO,
DEPARTAMENTOS.DESCRIPCION AS DEPARTAMENTO,
NOMINATYPE.DESCRIPCION
FROM 
PRENOMINAGEN
 INNER JOIN NOMINAUNI ON NOMINAUNI.IDNOMINAUNI=PRENOMINAGEN.IDNOMINAGEN
 INNER JOIN TRABAJADOR ON TRABAJADOR.IDTRABAJADOR=PRENOMINAGEN.IDTRABAJADOR
 INNER JOIN CARGO ON CARGO.IDCARGO=TRABAJADOR.IDCARGO
 INNER JOIN DEPARTAMENTOS ON DEPARTAMENTOS.IDDEPARTAMENTOS=TRABAJADOR.IDDEPARTAMENTOS
 INNER JOIN NOMINATYPE ON NOMINATYPE.IDNOMINA=PRENOMINAGEN.IDNOMINATYPE
